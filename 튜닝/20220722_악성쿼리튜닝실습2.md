## 케이스 정리

```
-- BEFORE
-- 문제점
-- 10,001 ~ 10,100 사원번호 범위의 통계 데이터가 필요한거니까 해당 사원번호와 관련된 급여 데이터만 가져오면 되는데 모든 급여 데이터를 가져오고 있음.
SELECT 사원.사원번호,
       급여.평균연봉,
       급여.최고연봉,
       급여.최저연봉
  FROM 사원,
       ( SELECT 사원번호,
                ROUND(AVG(연봉), 0) 평균연봉,
                ROUND(MAX(연봉), 0) 최고연봉,
                ROUND(MIN(연봉), 0) 최저연봉
           FROM 급여
          GROUP BY 사원번호
       ) 급여
 WHERE 사원.사원번호 = 급여.사원번호
   AND 사원.사원번호 BETWEEN 10001 AND 10100;
-- AFTER
-- 성능 개선: 1.462s -> 2ms
SELECT STRAIGHT_JOIN
	   사원.사원번호,
       ROUND(AVG(연봉), 0) 평균연봉,
       ROUND(MAX(연봉), 0) 최고연봉,
       ROUND(MIN(연봉), 0) 최저연봉
  FROM 급여
 INNER JOIN 사원
    ON 사원.사원번호 = 급여.사원번호
 WHERE 급여.사원번호 BETWEEN 10001 AND 10100
 GROUP BY 급여.사원번호;
```

```
-- 쿼리 목적
-- BEFORE
-- 문제점
-- 그루핑, 내림차순 정렬을 위해서 임시 테이블 생성.
-- 정렬 연산을 위해서 filesort 수행.
-- 10건의 데이터를 출력하기 위해서 수십만건의 데이터를 조인, 그룹핑 및 정렬 연산을 하는 행위는 매우 비효율적임.
SELECT 사원.사원번호
       ,사원.이름
       ,사원.성
       ,사원.입사일자
       ,SUM(급여.연봉) 연봉
  FROM 사원, 급여
 WHERE 사원.사원번호 = 급여.사원번호 
   AND 사원.사원번호 BETWEEN 10001 AND 50000
 GROUP BY 사원.사원번호
 ORDER BY SUM(급여.연봉) DESC
 LIMIT 150, 10;
-- AFTER
-- 성능 개선: 335ms -> 5ms
SELECT 사원.사원번호
       ,사원.이름
       ,사원.성
       ,사원.입사일자
       ,급여.연봉
  FROM (SELECT 사원번호
               ,SUM(급여.연봉) 연봉
  	      FROM 급여
  	     WHERE 사원번호 BETWEEN 10001 AND 50000
  	     GROUP BY 사원번호
  	     LIMIT 150, 10) 급여, 사원
 WHERE 사원.사원번호 = 급여.사원번호
 ORDER BY 급여.연봉 DESC;
```

```
-- 쿼리 목적
-- BEFORE
-- 문제점
-- 출력하지 않는 부서관리자 데이터를 가져와서 조인하고 있음
SELECT COUNT(사원번호) 카운트
  FROM (SELECT 사원.사원번호
               ,부서관리자.부서번호
          FROM (SELECT *
                  FROM 사원
                 WHERE 사원번호 > 300000
                   AND 성별 = 'M') 사원
          LEFT JOIN 부서관리자
                 ON 사원.사원번호 = 부서관리자.사원번호) 서브쿼리;
-- AFTER
-- 성능 개선: 113ms -> 40ms
SELECT COUNT(사원번호) 카운트
  FROM 사원
 WHERE 사원번호 > 300000
   AND 성별 = 'M';
```

```
-- 쿼리 목적
-- BEFORE
-- 문제점
-- 수신만 데이터를 먼저 조인시키고 중복 제거 행위를 수행하고 있어서 비효율적임.
EXPLAIN
SELECT DISTINCT 매핑.부서번호
  FROM 부서관리자 관리자,
       부서사원_매핑 매핑
 WHERE 관리자.부서번호 = 매핑.부서번호 
 ORDER BY 매핑.부서번호;
-- AFTER
-- 성능 개선: 498ms -> 2ms
SELECT 매핑.부서번호
  FROM (SELECT DISTINCT 부서번호
          FROM 부서사원_매핑 매핑) 매핑
 WHERE EXISTS (SELECT 1
                 FROM 부서관리자 관리자
                WHERE 부서번호 = 매핑.부서번호);
```

```
-- 쿼리 목적
-- BEFORE
-- 문제점
-- 인덱스를 적용하지 않고 데이터를 찾아서 테이블 풀 스캔이 일어나고 있음.
SELECT *
  FROM 사원
 WHERE 이름 = 'Georgi'
   AND 성 = 'Wielonsky';
-- AFTER
-- 성능 개선: 150ms -> 2ms
ALTER TABLE 사원 ADD INDEX I_사원_성_이름 (성,이름); 
SELECT *
  FROM 사원
 WHERE 성 = 'Wielonsky'
   AND 이름 = 'Georgi';
```
